float update_PI(PI_Controller *pi, float feedback, float dt) {
    float error = pi->setpoint - feedback;

    // Proportional term
    float P = pi->Kp * error;

    // Integral term
    pi->integral += error * dt;

    // Simple Anti-Windup: Clamp the integral term
    if (pi->integral > 50.0f) pi->integral = 50.0f;
    if (pi->integral < -50.0f) pi->integral = -50.0f;

    float I = pi->Ki * pi->integral;

    float output = P + I;

    // Final Output Clamping (Safety)
    if (output > pi->output_limit) output = pi->output_limit;
    if (output < 0.05f) output = 0.05f;

    return output;
}

uint8_t read_hall_sensors(void) {
    uint8_t state = 0;

    // Read each pin and shift it to the correct bit (Bit 2, Bit 1, Bit 0)
    state |= (HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_0) << 2); // Sensor A to Bit 0
    state |= (HAL_GPIO_ReadPin(GPIOC, GPIO_PIN_2) << 1); // Sensor B to Bit 1
    state |= (HAL_GPIO_ReadPin(GPIOC, GPIO_PIN_3) << 0); // Sensor C to Bit 2

    return state; // Passes a 3-bit value (0-7) to your next function
}

	// STEP 1: Global Safety - Disable Main Output temporarily
	    __HAL_TIM_MOE_DISABLE(&htim1);

    switch (hes_code)
    {

        case 0b101: // Phase B high, Phase A switching, Phase C (CH4) floating
                    // Phase A Switching (CH1)
        			TIM1->CCR1 = ((1-duty_cycle)*ARR);
        			HAL_TIMEx_PWMN_Start(&htim1, TIM_CHANNEL_1);
        	        HAL_TIM_PWM_Start(&htim1, TIM_CHANNEL_1);

                    // Phase B High (CH2)
        	        TIM1->CCR2 = ARR;
                	HAL_TIMEx_PWMN_Start(&htim1, TIM_CHANNEL_2);
                	HAL_TIM_PWM_Start(&htim1, TIM_CHANNEL_2);

                    // Phase C Floating (CH4)
                	HAL_TIMEx_PWMN_Stop(&htim1, TIM_CHANNEL_4);
                	HAL_TIM_PWM_Stop(&htim1, TIM_CHANNEL_4);
                    break;

        case 0b100: // Phase C (CH4) high, Phase A switching, Phase B floating
                    // Phase A Switching (CH1)
        			TIM1->CCR1 = ((1-duty_cycle)*ARR);
        			HAL_TIMEx_PWMN_Start(&htim1, TIM_CHANNEL_1);
        			HAL_TIM_PWM_Start(&htim1, TIM_CHANNEL_1);
                    // Phase B Floating (CH2)
                	HAL_TIMEx_PWMN_Stop(&htim1, TIM_CHANNEL_2);
                	HAL_TIM_PWM_Stop(&htim1, TIM_CHANNEL_2);
                    // Phase C High (CH4)
                	TIM1->CCR4 = ARR;
                	HAL_TIMEx_PWMN_Start(&htim1, TIM_CHANNEL_4);
                	HAL_TIM_PWM_Start(&htim1, TIM_CHANNEL_4);
                    break;

        case 0b110: // Phase C (CH4) high, Phase B switching, Phase A floating
                    // Phase A Floating (CH1)
                	HAL_TIMEx_PWMN_Stop(&htim1, TIM_CHANNEL_1);
                	HAL_TIM_PWM_Stop(&htim1, TIM_CHANNEL_1);
                    // Phase B Switching (CH2)
                	TIM1->CCR2 = ((1-duty_cycle)*ARR);
                	HAL_TIMEx_PWMN_Start(&htim1, TIM_CHANNEL_2);
                	HAL_TIM_PWM_Start(&htim1, TIM_CHANNEL_2);
                    // Phase C High (CH4)
                	TIM1->CCR4 = ARR;
                	HAL_TIMEx_PWMN_Start(&htim1, TIM_CHANNEL_4);
                	HAL_TIM_PWM_Start(&htim1, TIM_CHANNEL_4);
                    break;

        case 0b010: // Phase A high, Phase B switching, Phase C (CH4) floating
                    // Phase A High (CH1)
                	TIM1->CCR1 = ARR;
                	HAL_TIMEx_PWMN_Start(&htim1, TIM_CHANNEL_1);
                	HAL_TIM_PWM_Start(&htim1, TIM_CHANNEL_1);
                    // Phase B Switching (CH2)
                	TIM1->CCR2 = ((1-duty_cycle)*ARR);
                	HAL_TIMEx_PWMN_Start(&htim1, TIM_CHANNEL_2);
                	HAL_TIM_PWM_Start(&htim1, TIM_CHANNEL_2);
                    // Phase C Floating (CH4)
                	HAL_TIMEx_PWMN_Stop(&htim1, TIM_CHANNEL_4);
                	HAL_TIM_PWM_Stop(&htim1, TIM_CHANNEL_4);
                    break;

        case 0b011: // Phase A high, Phase C (CH4) switching, Phase B floating
                    // Phase A High (CH1)
        			TIM1->CCR1 = ARR;
        	        HAL_TIMEx_PWMN_Start(&htim1, TIM_CHANNEL_1);
        	        HAL_TIM_PWM_Start(&htim1, TIM_CHANNEL_1);
                    // Phase B Floating (CH2)
                	HAL_TIMEx_PWMN_Stop(&htim1, TIM_CHANNEL_2);
                	HAL_TIM_PWM_Stop(&htim1, TIM_CHANNEL_2);
                    // Phase C Switching (CH4)
                	TIM1->CCR4 = ((1-duty_cycle)*ARR);
                	HAL_TIMEx_PWMN_Start(&htim1, TIM_CHANNEL_4);
                	HAL_TIM_PWM_Start(&htim1, TIM_CHANNEL_4);
                    break;

        case 0b001: // Phase B high, Phase C (CH4) switching, Phase A floating
                    // Phase A Floating (CH1) //works expect this shit
                	HAL_TIMEx_PWMN_Stop(&htim1, TIM_CHANNEL_1);
                	HAL_TIM_PWM_Stop(&htim1, TIM_CHANNEL_1);
                    // Phase B High (CH2)
                	TIM1->CCR2 = ARR;
                	HAL_TIMEx_PWMN_Start(&htim1, TIM_CHANNEL_2);
                	HAL_TIM_PWM_Start(&htim1, TIM_CHANNEL_2);
                    // Phase C Switching (CH4)
                	TIM1->CCR4 = ((1-duty_cycle)*ARR);
                	HAL_TIMEx_PWMN_Start(&htim1, TIM_CHANNEL_4);
                	HAL_TIM_PWM_Start(&htim1, TIM_CHANNEL_4);
                    break;

        default:
            // Emergency Stop: Force everything inactive

            break;
    }

        // STEP 4: Global Safety - Re-enable Main Output
        __HAL_TIM_MOE_ENABLE(&htim1);
}

    	    }

    	    }

    	    }
    	}
    }
